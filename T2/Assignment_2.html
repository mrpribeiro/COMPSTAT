Dataset description

The original data was acquired from critically-ill patients during hospital care at the Beth Israel Deaconess Medical Centre (Boston, MA, USA). Two annotators manually annotated individual breaths in each recording using the impedance respiratory signal. The 53 recordings within the dataset, each of 8-minute duration, each contain:

  • Physiological signals include the photoplethysmogram (PPG), impedance respiratory signal, and electrocardiogram (ECG). These are sampled at 125 Hz.
  • Physiological parameters, such as the heart rate (HR), respiratory rate (RESP), and blood oxygen saturation level (SpO2). These are sampled at 1 Hz.
  • Fixed parameters, such as age and gender
  • Manual annotations of breaths

1. Download data and create a data frame containing the mean HR and age for each subject.

# Create empty data frame to store mean HR and age for each subject
data_summary <- data.frame(subject = character(),
                           mean_HR = numeric(),
                           age = numeric(),
                           stringsAsFactors = FALSE)
# Loop through each subject's files
for (i in 1:53) {
  # Create file links for numerics and fixed data
  numerics_file <- paste0("https://physionet.org/files/bidmc/1.0.0/bidmc_csv/bidmc_", sprintf("%02d", i), "_Numerics.csv?download")
  fix_file <- paste0("https://physionet.org/files/bidmc/1.0.0/bidmc_csv/bidmc_", sprintf("%02d", i), "_Fix.txt?download")
  
  # Read the numerics data
  numerics_data <- read.csv(numerics_file)
  
  # Read the fixed data
  fix_data <- readLines(fix_file, warn = FALSE)
  
  # Calculate mean HR
  mean_HR <- mean(numerics_data$HR, na.rm = TRUE)
  
  # Age extraction from fixed data: locate the line starting with "Age" dynamically
  age_line_idx <- grep("^\\s*Age", fix_data, ignore.case = TRUE)
    
  if (length(age_line_idx) > 0) {
    # Extract numeric value safely using regex on the identified line
    age_str <- sub(".*?([0-9]+).*", "\\1", fix_data[age_line_idx[1]])
    age <- as.integer(age_str)
  } else {
    # Safe fallback if Age is missing from the file
    age <- NA 
  }
  
  # Append to the summary data frame
  data_summary <- rbind(data_summary, data.frame(subject = paste0("bidmc_",
                                                        sprintf("%02d", i)),
                                                        mean_HR = mean_HR,
                                                        age = age))
}

3. Apply the expectation-maximization algorithm with 2 and 3 latent classes.

- Manual implementation -

> Soft

2 latent classes

# iteration 0
m <- 2 # number of latent classes
eps <- 0.01 # convergence error
itera <- 0 # iteration counter
maxitera <- 100 # maximum number of iterations
set.seed(523) # for reproducibility
param.mean <- sample(data_summary$mean_HR, m) # initial mean values for the latent classes, randomly sampled from the data
param.sd <- rep(sd(data_summary$mean_HR) / m, m) # initial standard deviation values for the latent classes, set to a fraction of the overall standard deviation of the data

for (itera in 0:maxitera) { # EM algorithm loop
  names(param.mean) <- paste("mean",1:m) # assign names to the mean parameters for better readability
  names(param.sd) <- paste("sd",1:m) # assign names to the standard deviation parameters for better readability
  print(param.mean) # print the current mean parameters for each latent class
  print(param.sd) # print the current standard deviation parameters for each latent class

  ## Expectation-Maximization - E-Step

  # data points (mean HR for each subject)
  xx <- data_summary$mean_HR 
  # calculate the density of each data point under each latent class distribution
  densities <- sapply(1:m, function(i){dnorm(xx,mean=param.mean[i],sd=param.sd[i])})
  # calculate the relevance (responsibility) of each latent class for each data point by normalizing the densities
  relevance <- t(apply(densities, 1, function(l){l/sum(l)})) 

  ## Expectation-Maximization - M-Step soft

  # update the mean parameters for each latent class by calculating the weighted average of the data points, where the weights are given by the relevance of each latent class for each data point
  param.mean.new <- sapply(1:m, function(j){sum(data_summary$mean_HR*relevance[,j])/sum(relevance[,j])})
  # update the standard deviation parameters for each latent class by calculating the weighted standard deviation of the data points, where the weights are given by the relevance of each latent class for each data point
  param.sd.new <- sapply(1:m, function(j){sqrt(sum(relevance[,j]*(data_summary$mean_HR-param.mean[j])^2)/sum(relevance[,j]))})


  ## Expectation-Maximization - Convergence

  # calculate the difference between the new and old mean parameters to check for convergence
  cat("Iteration",itera,"evolution:",dif <- sum(abs(param.mean.new - param.mean)), "\n")
  # if the difference is less than the convergence threshold, exit the loop
  if (dif < eps) break
  param.mean <- param.mean.new # update the mean parameters for the next iteration
  param.sd <- param.sd.new # update the standard deviation parameters for the next iteration
}

set.seed(523) # for reproducibility
param.mean <- sample(data_summary$mean_HR, m) # initial mean values for the latent classes, randomly sampled from the data
param.sd <- rep(sd(data_summary$mean_HR) / m, m) # initial standard deviation values for the latent classes, set to a fraction of the overall standard deviation of the data

for (itera in 0:maxitera) { # EM algorithm loop
  names(param.mean) <- paste("mean",1:m) # assign names to the mean parameters for better readability
  names(param.sd) <- paste("sd",1:m) # assign names to the standard deviation parameters for better readability
  print(param.mean) # print the current mean parameters for each latent class
  print(param.sd) # print the current standard deviation parameters for each latent class

  ## Expectation-Maximization - E-Step

  # data points (mean HR for each subject)
  xx <- data_summary$mean_HR
  # calculate the density of each data point under each latent class distribution
  densities <- sapply(1:m, function(i){dnorm(xx,mean=param.mean[i],sd=param.sd[i])})
  # calculate the relevance (responsibility) of each latent class for each data point by normalizing the densities 
  relevance <- t(apply(densities, 1, function(l){l/sum(l)})) 

  ## Expectation-Maximization - M-Step crisp

  # assign each data point to the latent class with the highest relevance (crisp assignment)
  attrib <- t(apply(relevance, 1, function(l){replace(rep(0,m), which.max(l)[1], 1)}))
  # update the mean parameters for each latent class by calculating the average of the data points that are assigned to that class
  param.mean.new <- sapply(1:m, function(j){mean(data_summary$mean_HR[attrib[,j]==1])})
  # update the standard deviation parameters for each latent class by calculating the standard deviation of the data points that are assigned to that class 
  param.sd.new <- sapply(1:m, function(j){sd(data_summary$mean_HR[attrib[,j]==1])}) 

  ## Expectation-Maximization - Convergence

  # calculate the difference between the new and old mean parameters to check for convergence
  cat("Iteration",itera,"evolution:",dif <- sum(abs(param.mean.new - param.mean)), "\n")
  # if the difference is less than the convergence threshold, exit the loop
  if (dif < eps) break 
  param.mean <- param.mean.new # update the mean parameters for the next iteration
  param.sd <- param.sd.new # update the standard deviation parameters for the next iteration
}
